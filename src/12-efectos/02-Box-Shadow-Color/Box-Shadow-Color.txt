shadow-black  |  --tw-shadow-color: #000;
shadow-white  |  --tw-shadow-color: #fff;
shadow-slate-50  |  --tw-shadow-color: #f8fafc;
shadow-slate-100  |  --tw-shadow-color: #f1f5f9;
shadow-slate-200  |  --tw-shadow-color: #e2e8f0;
